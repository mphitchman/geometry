\begin{align*} \amp \int_0^{2\pi} \int_0^x \frac{4r}{(1-r^2)^2}~drd\theta\\ \amp = \bigg(\int_0^{2\pi}~d\theta\bigg)\bigg(\int_1^{1-x^2}\frac{-2}{u^2}~du\bigg)\\ \amp = 2\pi\bigg[\frac{2}{u} \bigg|_1^{1-x^2}\bigg]\\ \amp = 2\pi \bigg[ \frac{2}{1-x^2} - 2\bigg]\\ \amp = 4\pi \frac{x^2}{1-x^2}. \end{align*}
