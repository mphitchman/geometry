The third side has length

\begin{equation*} c = s\ln\left(\frac{s+|T(r)|}{s-|T(r)|}\right) \end{equation*}

where

\begin{equation*} T(z) = \frac{z - me^{i\gamma}}{1 + kme^{-i\gamma}z} \end{equation*}

is a transformation in \(G_k\) taking \(me^{i\gamma}\) to 0. Thus,

\begin{align*} \cosh(\sqrt{|k|}c) \amp= \cosh\left(\ln\left(\frac{s+|T(r)|}{s-|T(r)|}\right)\right)\\ \amp= \frac{(s+|T(r)|)^2+(s-|T(r)|)^2}{2(s+|T(r)|)(s-|T(r)|)}\\ \amp= \frac{s^2 + |T(r)|^2}{s^2-|T(r)|^2}\text{.} \end{align*}

Now,

\begin{align*} |T(r)|^2 \amp= \frac{|r - me^{i\gamma}|^2}{|1 + kmre^{-i\gamma}|^2}\\ \amp= \frac{(r - me^{i\gamma})(r - me^{-i\gamma})}{(1+krme^{i\gamma})(1 + krme^{-i\gamma})}\\ \amp= \frac{r^2 + m^2 - 2rm\cos(\gamma)}{1+(krm)^2 + 2krm\cos(\gamma)}\text{.} \end{align*}

Then,

\begin{align*} \cosh(\sqrt{|k|}c) \amp= \frac{s^2 + |T(r)|^2}{s^2-|T(r)|^2}\\ \amp= \frac{s^2[1+(krm)^2 + 2krm\cos(\gamma)]+[r^2+m^2-2rm\cos(\gamma)]}{s^2[1+(krm)^2 + 2krm\cos(\gamma)]-[r^2+m^2-2rm\cos(\gamma)]}\\ \amp= \frac{[s^2 + (skrm)^2-2rm\cos(\gamma)]+[r^2 + m^2 - 2rm\cos(\gamma)]}{[s^2 + (skrm)^2-2rm\cos(\gamma)]-[r^2 + m^2 - 2rm\cos(\gamma)]}\\ \amp= \frac{s^2 + (skrm)^2 + r^2 + m^2 - 4rm\cos(\gamma)}{s^2 + (skrm)^2 - r^2 - m^2}\\ \amp= \frac{s^4 + (s^2krm)^2 + s^2r^2 + s^2m^2 - 4rms^2\cos(\gamma)}{s^4+(s^2krm)^2 - s^2r^2-s^2m^2}\\ \amp= \frac{(s^2 + r^2)(s^2+m^2) - (2rs)(2ms)\cos(\gamma)}{(s^2 - r^2)(s^2-m^2)}\\ \amp=\cosh(\sqrt{|k|}a)\cosh(\sqrt{|k|}b)-\sinh(\sqrt{|k|}a)\sinh(\sqrt{|k|}b)\cos(\gamma)\text{.} \end{align*}
