\begin{align*} {\cal L}(\boldsymbol{r}) \amp= \frac{4\pi s \frac{e^{\frac{r}{s}}-1}{e^{\frac{r}{s}}+1}}{1-\bigg(\frac{e^{\frac{r}{s}}-1}{e^{\frac{r}{s}}+1}\bigg)^2}\\ \amp=4\pi s \frac{(e^{\frac{r}{s}}-1)(e^{\frac{r}{s}}+1)}{(e^{\frac{r}{s}}+1)^2-(e^{\frac{r}{s}}-1)^2}\\ \amp=4\pi s \frac{e^{\frac{2r}{s}}-1}{4e^{\frac{r}{s}}}\\ \amp= 2\pi s \frac{e^{\frac{r}{s}}-e^{-\frac{r}{s}}}{2}\\ \amp= 2\pi s \sinh\left(\frac{r}{s}\right) \end{align*}
