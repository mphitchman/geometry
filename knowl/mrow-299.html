\begin{align*} 4\pi \frac{a}{1-a^2} \amp= 4\pi\frac{\frac{e^r - 1}{e^r + 1}}{1-\frac{(e^r - 1)^2}{(e^r + 1)^2}}\\ \amp= 4\pi\frac{(e^r-1)(e^r+1)}{(e^r+1)^2-(e^r-1)^2}\\ \amp= 4\pi\frac{e^{2r}- 1}{(e^{2r}+2e^r+1)-(e^{2r}-2e^r+1)}\\ \amp= 4\pi\frac{e^{2r}-1}{4e^r}\\ \amp= 2\pi\frac{e^r - e^{-r}}{2}\\ \amp= 2\pi\sinh(r) \end{align*}
