1. Suppose \(z = x+yi\) and \(w = s+ti\text{.}\) Then
    \begin{align*} |zw| \amp = \sqrt{(xs-yt)^2 + (xt+ys)^2}\\ \amp = \sqrt{x^2s^2 - 2xyst + y^2t^2 + x^2t^2 + 2xyst + y^2s^2}\\ \amp = \sqrt{x^2(s^2+t^2)+y^2(s^2 + t^2)} \\ \amp = \sqrt{(x^2 + y^2)(s^2+t^2)} \\ \amp = |z|\cdot |w|\text{.} \end{align*}
  2. Suppose \(z = x+yi\) and \(w = s+ti\text{.}\) Then
    \begin{align*} \overline{z\cdot w} \amp = \overline{(xs-yt)+(ys+xt)i}\\ \amp = (xs-yt)-(ys+xt)i \end{align*}
    and
    \begin{align*} \overline{z}\cdot\overline{w} \amp = (x-yi)(s-ti)\\ \amp = (xs-yt)-(ys+xt)i\text{.} \end{align*}
    Thus, \(\overline{z\cdot w} = \overline{z}\cdot\overline{w}\text{.}\)
  3. Suppose \(z = x+yi\) and \(w = s+ti\text{.}\) Then
    \begin{align*} \overline{z+w}\amp =\overline{(x+s)+(y+t)i} \\ \amp = (x+s)-(y+t)i \\ \amp = (x-yi)+(s-ti) \\ \amp = \overline{z}+\overline{w} \text{.} \end{align*}
  4. Suppose \(z = x + yi\text{.}\) Then
    \begin{align*} z+\overline{z} \amp = (x+yi) + (x-yi) \\ \amp = 2x \\ \amp = 2\text{Re}(z)\text{.} \end{align*}
  5. Suppose \(z = x + yi\text{.}\) Then
    \begin{align*} z-\overline{z} \amp = (x+yi) - (x-yi) \\ \amp = 2yi \\ \amp = 2\text{Im}(z)i\text{.} \end{align*}
  6. Suppose \(z = x + yi\text{.}\) Then
    \begin{equation*} |\overline{z}| = \sqrt{x^2+(-y)^2} = \sqrt{x^2+y^2}=|z|\text{.} \end{equation*}
