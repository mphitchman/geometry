\begin{align*} {\cal L}(\boldsymbol{r}) \amp= \int_0^x \frac{2|\boldsymbol{r}^\prime(t)|}{1 + k|\boldsymbol{r}(t)|^2}~dt\\ \amp=\int_0^x \frac{2\cdot 1}{1 + kt^2}~dt\\ \amp= \frac{2}{\sqrt{k}}\int_0^x \frac{\sqrt{k}}{1+(\sqrt{k}t)^2}~dt\\ \amp=\frac{2}{\sqrt{k}}\int_0^{\sqrt{k}x}\frac{1}{1+u^2}~du\\ \amp= \frac{2}{\sqrt{k}}\arctan(\sqrt{k}x)\\ \amp= 2s \arctan\left(\frac{x}{s}\right) \end{align*}
