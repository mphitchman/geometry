Then

\begin{equation*} {\cal L}(\boldsymbol{s}(t)) = \int_a^b \frac{2|\boldsymbol{s}^\prime(t)|}{1 - |\boldsymbol{s}(t)|^2} ~dt\text{,} \end{equation*}

and the integrand equals

\begin{align*} \frac{2|\boldsymbol{s}^\prime(t)|}{1 - |\boldsymbol{s}(t)|^2} \amp= \frac{2\bigg|\frac{1-|\overline{z_0}|^2}{(1-\overline{z_0}\boldsymbol{r}(t))^2}\cdot \boldsymbol{r}^\prime(t)\bigg|}{1-\bigg|\frac{\boldsymbol{r}(t)-z_0}{1-\overline{z_0}\boldsymbol{r}(t)}\bigg|^2}\\ \amp= \frac{2|\boldsymbol{r}^\prime(t)|\cdot |1 - |z_0|^2|\cdot\frac{1}{|1-\overline{z_0} \boldsymbol{r}(t)|^2}}{\frac{|1-\overline{z_0}\boldsymbol{r}(t)|^2 - |\boldsymbol{r}(t) - z_0|^2}{|1-\overline{z_0}\boldsymbol{r}(t)|^2}}\\ \amp= \frac{2|\boldsymbol{r}^\prime(t)|\cdot |1 - |z_0|^2|}{|1-\overline{z_0}\boldsymbol{r}(t)|^2 - |\boldsymbol{r}(t) - z_0|^2}\\ \amp=\frac{2|\boldsymbol{r}^\prime(t)|\cdot |1 - |z_0|^2|}{(1 - \overline{z_0}\boldsymbol{r}(t)-z_0\overline{\boldsymbol{r}(t)} + |z_0|^2|\boldsymbol{r}(t)|^2)-(|\boldsymbol{r}(t)|^2 - \boldsymbol{r}(t)\overline{z_0}- \overline{\boldsymbol{r}(t)}z_0 + |z_0|^2)}\\ \amp=\frac{2|\boldsymbol{r}^\prime(t)|\cdot |1 - |z_0|^2|}{1 - |\boldsymbol{r}(t)|^2 - |z_0|^2 + |z_0|^2|\boldsymbol{r}(t)|^2}\\ \amp= \frac{2|\boldsymbol{r}^\prime(t)|\cdot |1 - |z_0|^2|}{(1-|z_0|^2)(1-|\boldsymbol{r}(t)|^2)}\\ \amp= \frac{2|\boldsymbol{r}^\prime(t)|}{1 - |\boldsymbol{r}(t)|^2}\text{.} \end{align*}
